df = pd.DataFrame(data)

# Calcular o IQR para cada coluna
Q1 = df.quantile(0.25)
Q3 = df.quantile(0.75)
IQR = Q3 - Q1

# Definir o intervalo para remover outliers
lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR

# Filtrar os dados removendo outliers
df_sem_outliers = df[~((df < lower_bound) | (df > upper_bound)).any(axis=1)]

print("DataFrame sem outliers:")
print(df_sem_outliers)
